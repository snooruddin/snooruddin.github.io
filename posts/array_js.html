Array

Javascript array is much like any other arrays of other programming languages. The Array object is a global object that is used in constructing arrays.

Basics 

                
                    // creating an array with values
                    var arrWithValues = [12, 43, 56, 76, 453, 23];

                    //creating an empty array
                    //you can insert data into an empty array using built-in methods like push(), shift() etc.
                    var arrWithoutValues = [];

                    //creating array with blank elements
                    var arr = new Array(3); //create array with 3 blank elements
                    var arr = Array(3)
                    console.log(arr) // [, , ,]
                    //you can add elements in this array using the indexes
                    arr[0] = 'imad';
                    arr[1] = 'sharna';
                    arr[2] = 'daisy';
                    console.log(arr); // ['imad', 'sharna', 'daisy']

                    //access the elements inside the array
                    var first = arrWithValues[0] // to access the 0'th element
                    var second = arrWithValues[1] // to access the 1st element


                    //array of objects or most commonly knows as JSON
                    //the elements inside {} means that they are anonymous objects

                    //Notice the anonymous objects are separated by comma
                    var arrWithObjects = [
                        {
                            id : 1,

                        },
                        {
                            id : 2,

                        },

Array.length 

                
                    //Array.lenth returns length of an array
                    // Yeah, no wonder
                    //length returns the number of elements in the array
                    var arr = ['imad', 'falguni', 'daisy'];
                    var lengthOfArray = arr.length; // lengthOfArray = 3

                    //the last element in the array is in the length - 1 index
                    var lastElement = arr[arr.length - 1]; // lastElement = 'daisy'

                    //Important : length in array is a writable property
                    arr.length = 5; //this is totally valid
                    console.log(arr);  // ['imad', 'falguni', 'daisy', undefined, undefined]

                    //IMPORTANT: this is a very good way to empty an array that has not been referenced otherwise
                    arr.length = 0; //deletes all the elements in the array
                    // another good way is like below:
                    var arr = []; // this basically redeclares the array as an empty array

Array.from()

This creates a new array instance from another array or array like iterable objects. This can create objects from the following:

1. Array like objects( objects with length property and indexed elements )

2. Iterable objects such as map or set

An arrow function can also be used be used with from() which will operate on all the elements of the newly created array

                
                    // array from a string
                    var arrFromStr = Array.from("foobar");
                    console.log(arrFromSet);  // the result is: ['f','o','o','b','a','r']

                    // array from a set
                    var s = new Set([1, 2, 3, 4]);
                    var arrFromSet = Array.from(s);
                    console.log(arrFromSet);  //the result is : [1, 2, 3, 4];

                    // array from a map
                    var m = new Map([ [1,2], [3,4], [5,6] ]);
                    var arrFromMap = Array.from(m);
                    console.log(arrFromMap)  //the result is : [ [1,2], [3,4], [5,6] ]

                    //using a arrow function with from()
                    var arrWithArrow = Array.from([1, 2, 3, 4], (x) =>  x * x );
                    console.log(arrWithArrow);  // the result is : [ 1, 4, 9, 16 ]

Array.isArray()

Determines whether the passed value is an array. Returns true if passed value is an array and false otherwise. 

                
                    var arr = [1, 2, 3];
                    var str = 'foobar';

                    var result = Array.isArray(arr); // result = true
                    result = Array.isArray(str); // result = false
                    result = Array.isArray(NaN); // result = false
                    result = Array.isArray(undefined) // result = false;

                    //IMPORTANT: isArray() is preferred over instanceOf in checking whether object is array

Array.of()

This is just another array constructor. This method creates an array with the arguments passed to it. 

                
                    var arr = Array.of(1, 2, 3);
                    console.log(arr); // [1, 2, 3]

Array.prototype.concat()

This methods concat one or more arrays with the array that called the method. This method doesn't alter any of the arrays and returns the new array. 

                
                    var arr = [1, 2, 3];
                    var add = [4, 5, 6];

                    var result = arr.concat(add);
                    console.log(result); // [1, 2, 3, 4, 5, 6];

                    //adding multiple arrays
                    result = arr.concat([7,8], [9], [0]);
                    console.log(result); // [1, 2, 3, 4, 5, 6, 7, 8, 9, 0]

                    //adding array inside array
                    add.concat([ [8, 9] ]);
                    console.log(add); //[4, 5, 6, 8, 9]
